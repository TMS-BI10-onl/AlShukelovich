SELECT MANADGER_ID, FIRST_NAME, LAST_NAME, COUNT(1) AS ABC
FROM EMPLOYEES
GROUP BY MANADGER_ID, FIRST_NAME, LAST_NAME
HAVING COUNT(1)>6

2. 
SELECT DEPARTMENT_NAME, MIN(MIN_SALARY-MIN_SALARY/100*COMISSION_PCT) AS MinSalary, MAX(MAX_SALARY-MAX_SALARY/100*COMISSION_PCT) AS MaxSalary
FROM EMPLOYEES E LEFT JOIN JOBS J ON E.JOB_ID=J.JOB_ID
                 LEFT JOIN DEPARTMENTS D ON E.DEPARTMENT_ID=D.DEPARTMENT
GROUP BY DEPARTMENT_NAME

3.
SELECT TOP 1 WITH TIES R.REGION_NAME, COUNT(1) AS ABC
FROM EMPLOYEES E LEFT JOIN DEPARTMENTS D ON E.DEPARTMENT_ID=D.DEPARTMENT_ID
                 LEFT JOIN LOCATIONS L ON D.LOCATION_ID=L.LOCATION_ID
                 LEFT JOIN COUNTRIES C ON L.COUNTRY_ID=C.COUNTRY_ID
                 LEFT JOIN REGIONS R ON C.REGION_ID=R.REGION_ID
GROUP BY R.REGION_NAME
ORDER BY ABC DESC

4.
SELECT DEPARTMENT_NAME, ABS(AVG(SALARY)-(SELECT AVG(SALARY) FROM EMPLOYEES))/
                        (SELECT AVG(SALARY) FROM EMPLOYEES)*100
FROM EMPLOYEE E LEFT JOIN DEPARTMENTS D ON E.DEPARTMENT_ID=D.E.DEPARTMENT_ID
GROUP BY DEPARTMENT_NAME

5.
SELECT E.FIRST_NAME, E.LAST_NAME, D.DEPARTMENT_NAME
FROM EMPLOYEES E LEFT JOIN DEPARTMENTS D ON E.DEPARTMENT_ID=D.DEPARTMENT_ID
                 LEFT JOIN JOB_HISTORY JH ON D.DEPARTMENT_ID=JH.DEPARTMENT_ID
WHERE ((YEAR(DATEADD(YEAR,-10,GETDATE()))-YEAR(START_DATE))>10 AND YEAR(END_DATE) IS NULL)
      OR (YEAR(START_DATE)-YEAR(END_DATE))>10

SELECT *
FROM
